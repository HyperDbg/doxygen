229 //
230 // In order to enable External Interrupt Exiting we have to set
231 // PIN_BASED_VM_EXECUTION_CONTROLS_EXTERNAL_INTERRUPT in vmx
232 // pin-based controls (PIN_BASED_VM_EXEC_CONTROL) and also
233 // we should enable VM_EXIT_ACK_INTR_ON_EXIT on vmx vm-exit
234 // controls (VMCS_CTRL_VMEXIT_CONTROLS), also this function might not
235 // always be successful if the guest is not in the interruptible
236 // state so it wait for and interrupt-window exiting to re-inject
237 // the interrupt into the guest
238 //
239 if (VCpu->EnableExternalInterruptsOnContinue ||
240 VCpu->EnableExternalInterruptsOnContinueMtf)
241 {
242 //
243 // Ignore the interrupt as it's suppressed supressed because of instrumentation step-in
244 //
245
246 //
247 // During developing HyperDbg, we realized that if we just ignore the interrupts completely
248 // while we are waiting on 'i' instrumentation step-in command, then the serial device becomes
249 // unresposive, to solve this issue we hold the details of interrupts so we can re-inject
250 // and process them when we decide to continue the debuggee (guest interrupt windows is open)
251 // this way, the serial device works normally and won't become unresponsive
252 //
253 IdtEmulationInjectInterruptWhenInterruptWindowIsOpen(VCpu, InterruptExit);
254
255 //
256 // avoid incrementing rip
257 //
258 HvSuppressRipIncrement(VCpu);
259 }
260
261 else if (InterruptExit.Valid && InterruptExit.InterruptionType == INTERRUPT_TYPE_EXTERNAL_INTERRUPT)
262 {
263 __vmx_vmread(VMCS_GUEST_RFLAGS, &GuestRflags);
264 __vmx_vmread(VMCS_GUEST_INTERRUPTIBILITY_STATE, &InterruptibilityState);
265
266 //
267 // External interrupts cannot be injected into the
268 // guest if guest isn't interruptible (e.g.: guest
269 // is blocked by "mov ss", or EFLAGS.IF == 0).
270 //
271 Interruptible = GuestRflags.InterruptEnableFlag && !InterruptibilityState.BlockingByMovSs;
272
273 if (Interruptible)

275 //
276 // Re-inject the interrupt/exception
277 //
278 EventInjectInterruptOrException(InterruptExit);
279 }
280 else
281 {
282 //
283 // We can't inject interrupt because the guest's state is not interruptible
284 // we have to queue it an re-inject it when the interrupt window is opened!
285 //

287

292 }
