269 //
270 // In order to enable External Interrupt Exiting we have to set
271 // PIN_BASED_VM_EXECUTION_CONTROLS_EXTERNAL_INTERRUPT in vmx
272 // pin-based controls (PIN_BASED_VM_EXEC_CONTROL) and also
273 // we should enable VM_EXIT_ACK_INTR_ON_EXIT on vmx vm-exit
274 // controls (VMCS_CTRL_VMEXIT_CONTROLS), also this function might not
275 // always be successful if the guest is not in the interruptible
276 // state so it wait for and interrupt-window exiting to re-inject
277 // the interrupt into the guest
278 //
279 if (VCpu->EnableExternalInterruptsOnContinue ||
280 VCpu->EnableExternalInterruptsOnContinueMtf)
281 {
282 //
283 // Ignore the interrupt as it's suppressed supressed because of instrumentation step-in
284 //
285
286 //
287 // During developing HyperDbg, we realized that if we just ignore the interrupts completely
288 // while we are waiting on 'i' instrumentation step-in command, then the serial device becomes
289 // unresposive, to solve this issue we hold the details of interrupts so we can re-inject
290 // and process them when we decide to continue the debuggee (guest interrupt windows is open)
291 // this way, the serial device works normally and won't become unresponsive
292 //
293 IdtEmulationInjectInterruptWhenInterruptWindowIsOpen(VCpu, InterruptExit);
294
295 //
296 // avoid incrementing rip
297 //
298 HvSuppressRipIncrement(VCpu);
299 }
300 else if (InterruptExit.Valid && InterruptExit.InterruptionType == INTERRUPT_TYPE_EXTERNAL_INTERRUPT)
301 {
302 __vmx_vmread(VMCS_GUEST_RFLAGS, &GuestRflags);
303 __vmx_vmread(VMCS_GUEST_INTERRUPTIBILITY_STATE, &InterruptibilityState);
304
305 //
306 // External interrupts cannot be injected into the
307 // guest if guest isn't interruptible (e.g.: guest
308 // is blocked by "mov ss", or EFLAGS.IF == 0).
309 //
310 Interruptible = GuestRflags.InterruptEnableFlag && !InterruptibilityState.BlockingByMovSs;
311
312 if (Interruptible)
313 {
314 //
315 // Re-inject the interrupt/exception
316 //
317 EventInjectInterruptOrException(InterruptExit);
318 }
319 else
320 {
321 //
322 // We can't inject interrupt because the guest's state is not interruptible
323 // we have to queue it an re-inject it when the interrupt window is opened !
324 //
325 IdtEmulationInjectInterruptWhenInterruptWindowIsOpen(VCpu, InterruptExit);
326
327 //
328 // Enable Interrupt-window exiting.
329 //
330 HvSetInterruptWindowExiting(TRUE);
331 }
332
333 //
334 // avoid incrementing rip
335 //
336 HvSuppressRipIncrement(VCpu);
337 }
338 else
339 {
340 Interruptible = FALSE;
341
342 LogError("Err, why we are here? it's a vm-exit due to the external"
343 "interrupt and its type is not external interrupt? weird!");
344 }
345}
