645 //
646 // read the exit interruption information
647 //
648 __vmx_vmread(VMCS_VMEXIT_INTERRUPTION_INFORMATION, &InterruptExit);
649
650 //
651 // This type of vm-exit, can be either because of an !exception event,
652 // or it might be because we triggered APIC or X2APIC to generate an
653 // NMI, we want to halt the debuggee. We perform the checks here to
654 // avoid triggering an event for NMIs when the debuggee requested it
655 //
656 if (InterruptExit.InterruptionType == INTERRUPT_TYPE_NMI &&
657 InterruptExit.Vector == EXCEPTION_VECTOR_NMI)
658 {
659 //
660 // Check if we're waiting for an NMI on this core and if the guest is NOT in
661 // a instrument step-in ('i' command) routine
662 //
663 if (!VCpu->RegisterBreakOnMtf &&
664 VmxBroadcastNmiHandler(VCpu, FALSE))
665 {
666 return;
667 }
668 }
669
670 //
671 // *** When we reached here it means that this is not a NMI cause by guest,
672 // probably an event ***
673 //
674
675 //
676 // Triggering the pre-event
677 // As the context to event trigger, we send the vector or IDT Index
678 //
679 EventTriggerResult = VmmCallbackTriggerEvents(EXCEPTION_OCCURRED,
680 VMM_CALLBACK_CALLING_STAGE_PRE_EVENT_EMULATION,
681 InterruptExit.Vector,
682 &PostEventTriggerReq,
683 VCpu->Regs);
684
685 //
686 // Now, we check if the guest enabled MTF for debugging (instrumentation stepping)
687 // This is because based on Intel SDM :
688 // If the "monitor trap flag" VM-execution control is 1 and VM entry is
689 // injecting a vectored event, an MTF VM exit is pending on the instruction
690 // boundary before the first instruction following the VM entry
691 // and,
692 // If VM entry is injecting a pending MTF VM exit, an MTF VM exit is pending on the
693 // instruction boundary before the first instruction following the VM entry
694 // This is the case even if the "monitor trap flag" VM-execution control is 0
695 //
696 // So, we'll ignore the injection of Exception in this case
697 //
698 if (VCpu->RegisterBreakOnMtf)
699 {
700 return;
701 }
702
703 //
704 // Check whether we need to short-circuiting event emulation or not
705 //
706 if (EventTriggerResult != VMM_CALLBACK_TRIGGERING_EVENT_STATUS_SUCCESSFUL_IGNORE_EVENT)
707 {
708 //
709 // Handle exception (emulate or inject the event)
710 //
711 IdtEmulationHandleExceptionAndNmi(VCpu, InterruptExit);
712 }
713
714 //
715 // Check for the post-event triggering needs
716 //
717 if (PostEventTriggerReq)
718 {
719 VmmCallbackTriggerEvents(EXCEPTION_OCCURRED,
720 VMM_CALLBACK_CALLING_STAGE_POST_EVENT_EMULATION,
721 InterruptExit.Vector,
722 NULL,
723 VCpu->Regs);
