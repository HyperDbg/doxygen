176  __vmx_vmwrite(GUEST_RFLAGS, GuestRflags);

180  //

183  Cs.BASE = 0; // flat segment
184  Cs.LIMIT = (UINT32)~0; // 4GB limit

189  Ss.BASE = 0; // flat segment
190  Ss.LIMIT = (UINT32)~0; // 4GB limit
191  Ss.ATTRIBUTES.UCHARs = 0xC093; // G+DB+P+S+DPL0+Data
