142 //

144 }
145 }
146 else
147 {
148 AddressDetails->VirtualAddress = PhysicalAddressToVirtualAddressByProcessId(AddressDetails->PhysicalAddress, AddressDetails->ProcessId);
149
150 //
151 // We don't know a way for checking physical address validity
152 //
153 AddressDetails->KernelStatus = DEBUGGER_OPERATION_WAS_SUCCESSFUL;
154 }
