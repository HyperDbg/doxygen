733  }

736  // User-mode buffer is same as SearchMemRequest
737  //
738  UsermodeBuffer = SearchMemRequest;
739 
740  //
741  // We store the user-mode data in a seprate variable because
742  // we will use them later when we Zeroed the SearchMemRequest
743  //
744  AddressFrom = SearchMemRequest->Address;
745  AddressTo = SearchMemRequest->Address + SearchMemRequest->Length;

767  //
768  SearchAddressWrapper(SearchResultsStorage, SearchMemRequest, AddressFrom, AddressTo);
769 
770  //

774  // least SearchMemRequest bigger or equal to MaximumSearchResults * sizeof(UINT64)
775  // so we need to clear everything here, and also we should keep in mind that
776  // SearchMemRequest is no longer valid
777  //
778  RtlZeroMemory(SearchMemRequest, MaximumSearchResults * sizeof(UINT64));
779 
780  //
781  // It's time to move the results from our temporary buffer to the user-mode
782  // buffer, also there is something that we should check and that's the fact
783  // that we used aligned page addresses so the results should be checked to
784  // see whether the results are between the user's entered addresses or not
785  //
786  for (size_t i = 0; i < MaximumSearchResults; i++)
787  {
788  CurrentValue = SearchResultsStorage[i];
789 
790  if (CurrentValue == NULL)
791  {
792  //
793  // Nothing left to move
794  //
795  break;
796  }
797 
798  if (CurrentValue >= AddressFrom && CurrentValue <= AddressTo)
799  {
800  //

802  //

804  ResultsIndex++;
805  }
806  }
