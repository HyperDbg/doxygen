395  // The "use MSR bitmaps" VM-execution control is 0.
396  // The value of ECX is not in the ranges 00000000H - 00001FFFH and C0000000H - C0001FFFH
397  // The value of ECX is in the range 00000000H - 00001FFFH and bit n in read bitmap for low MSRs is 1,
398  // where n is the value of ECX.
399  // The value of ECX is in the range C0000000H - C0001FFFH and bit n in read bitmap for high MSRs is 1,
400  // where n is the value of ECX & 00001FFFH.
401  //
402 
403  //
404  // Execute WRMSR or RDMSR on behalf of the guest. Important that this
405  // can cause bug check when the guest tries to access unimplemented MSR
406  // even within the SEH block* because the below WRMSR or RDMSR raises
407  // #GP and are not protected by the SEH block (or cannot be protected
408  // either as this code run outside the thread stack region Windows
409  // requires to proceed SEH). Hypervisors typically handle this by noop-ing
410  // WRMSR and returning zero for RDMSR with non-architecturally defined
411  // MSRs. Alternatively, one can probe which MSRs should cause #GP prior
412  // to installation of a hypervisor and the hypervisor can emulate the
413  // results.
414  //
415 
416  //
417  // Check for sanity of MSR if they're valid or they're for reserved range for WRMSR and RDMSR
418  //
419  if ((GuestRegs->rcx <= 0x00001FFF) || ((0xC0000000 <= GuestRegs->rcx) && (GuestRegs->rcx <= 0xC0001FFF)) || (GuestRegs->rcx >= RESERVED_MSR_RANGE_LOW && (GuestRegs->rcx <= RESERVED_MSR_RANGE_HI)))
420  {
421  msr.Content = __readmsr(GuestRegs->rcx);
422  }
423 
424  //
425  // Check if it's EFER MSR then we show a false SCE state so the
426  // patchguard won't cause BSOD
427  //

429  {
430  EFER_MSR MsrEFER;
431  MsrEFER.Flags = msr.Content;
432  MsrEFER.SyscallEnable = TRUE;
433  msr.Content = MsrEFER.Flags;
434  }

437  GuestRegs->rdx = msr.High;
438 }
