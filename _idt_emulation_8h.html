226  // we should enable VM_EXIT_ACK_INTR_ON_EXIT on vmx vm-exit

231  //

234  {

236  __vmx_vmread(GUEST_INTERRUPTIBILITY_INFO, &InterruptibilityState);
237 
238  //
239  // External interrupts cannot be injected into the
240  // guest if guest isn't interruptible (e.g.: guest
241  // is blocked by "mov ss", or EFLAGS.IF == 0).
242  //
243  Interruptible = GuestRflags.InterruptEnableFlag && !InterruptibilityState.BlockingByMovSs;
244 
245  if (Interruptible)
246  {
247  //
248  // Re-inject the interrupt/exception
249  //
250  __vmx_vmwrite(VM_ENTRY_INTR_INFO, InterruptExit.Flags);
251 
252  //
253  // re-write error code (if any)
254  //
255  if (InterruptExit.ErrorCodeValid)
256  {
257  //
258  // Read the error code
259  //
260  __vmx_vmread(VM_EXIT_INTR_ERROR_CODE, &ErrorCode);
261 
262  //
263  // Write the error code
264  //
265  __vmx_vmwrite(VM_ENTRY_EXCEPTION_ERROR_CODE, ErrorCode);
266  }
267  }
268  else
269  {
270  //
271  // We can't inject interrupt because the guest's state is not interruptible
272  // we have to queue it an re-inject it when the interrupt window is opened !
273  //
274  for (size_t i = 0; i < PENDING_INTERRUPTS_BUFFER_CAPACITY; i++)
275  {
276  //
277  // Find an empty space
278  //
