638  return;
639  }
640 
641  //
642  // when we reached here, we know that it's a valid physical memory,
643  // so we change the structure and pass it as a virtual address to
644  // the search function
645  //
646  RealPhysicalAddress = SearchMemRequest->Address;
647 
648  //
649  // Change the start address
650  //
651  if (SearchMemRequest->ProcessId == PsGetCurrentProcessId())
652  {
653  SearchMemRequest->Address = PhysicalAddressToVirtualAddress(StartAddress);
654  EndAddress = PhysicalAddressToVirtualAddress(EndAddress);
655  }
656  else
657  {
658  SearchMemRequest->Address = PhysicalAddressToVirtualAddressByProcessId(StartAddress, SearchMemRequest->ProcessId);
659  EndAddress = PhysicalAddressToVirtualAddressByProcessId(EndAddress, SearchMemRequest->ProcessId);
660  }
661 
662  //
663  // Change the type of memory
664  //
665  SearchMemRequest->MemoryType = SEARCH_VIRTUAL_MEMORY;
666 
667  //
668  // Call the wrapper
669  //
670  PerformSearchAddress(AddressToSaveResults, SearchMemRequest, SearchMemRequest->Address, EndAddress);
671 
672  //
673  // Restore the previous state
674  //
675  SearchMemRequest->MemoryType = SEARCH_PHYSICAL_MEMORY;
676  SearchMemRequest->Address = RealPhysicalAddress;
677 
678  //
679  // Save the process id to avoid calling PsGetCurrentProcessId()
680  // multiple times
681  //
682  ProcId = PsGetCurrentProcessId();
683 
684  //
685  // Results should be ready (if any) in AddressToSaveResults
686  // we should convert it to a physical format as the search
687  // was on virtual format
688  //
689  for (size_t i = 0; i < MaximumSearchResults; i++)
690  {
691  CurrentValue = AddressToSaveResults[i];
692 
693  if (SearchMemRequest->ProcessId == ProcId && CurrentValue != NULL)
694  {
695  AddressToSaveResults[i] = VirtualAddressToPhysicalAddress(CurrentValue);
696  }
697  else if (CurrentValue != NULL)
698  {
699  AddressToSaveResults[i] = VirtualAddressToPhysicalAddressByProcessId(CurrentValue, SearchMemRequest->ProcessId);
700  }
701  else
702  {
703  break;
704  }
705  }
706 
707  return;
708  }
