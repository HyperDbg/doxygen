428  //
429 
430  //
431  // We are in kernel, so we should read MSR GS_BASE
432  // IA32_GS_BASE 0xC0000101
433  // IA32_KERNEL_GS_BASE 0xC0000102
434  // IA32_KERNEL_GS_BASE is currently user's gs as
435  // we are in the kernel-mode, if we need to intercept
436  // from user-mode then IA32_KERNEL_GS_BASE should be used
437  // but it's not for our case
438  //
439  MsrGsBase = __readmsr(IA32_GS_BASE);
440 
441  //
442  // Now, we have the gs base on MSR
443  // while in Windows gs:188 has the address
444  // of where the store current _ETHREAD, so
445  // we have to 188 to the gs base
446  //
447  MsrGsBase += 0x188;
448 
449  //

451  //

458

462  // we'll be able to change the dr7 of the guest on VMCS
463  //
464  HvSetLoadDebugControls(TRUE);

466 
467  //

471 
472  //

475  // registers and after disabling mov to dr (using vmcall),
476  // nothing is able to change debug registers so it's safe
477  //
478 
479  //

481  // read/write on the gs:188 as we intercept it on
482  // hypervisor side by exception bitmap on #DBs

484  // on Mov 2 Debug regs handler (vm-exit), but we set from here

486  //

496  // KiSaveProcessorControlState and KiRestoreProcessorControlState and
